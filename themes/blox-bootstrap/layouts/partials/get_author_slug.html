{{/* Takes an author's full name as input and returns a normalized slug. Example: "Salil S. Kanhere" -> "s-kanhere" */}} {{- $name_parts := split . " " -}} {{- if ge (len $name_parts) 1 -}} {{- $first_part := index $name_parts 0 -}} {{- $first_initial := substr $first_part 0 1 | lower -}} {{- $last_name := index $name_parts (sub (len $name_parts) 1) | lower -}} {{- printf "%s-%s" $first_initial $last_name -}} {{- end -}}
